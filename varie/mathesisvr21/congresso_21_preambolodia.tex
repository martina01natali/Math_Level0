  \def \larghezzaa{#1}
  \def \larghezzab{#2}
  \def \testoa{#3}
  \def \testob{#4}
  \begin{columns}
  \begin{column}{\larghezzaa\textwidth}
    \testoa
  \end{column}
  \begin{column}{\larghezzab\textwidth}
    \testob
  \end{column}
  \end{columns}
}
