\newcommand{\arcofunzione}[4]{% Traccia una freccia tra due nodi
  \def \posa{#1}
  \def \labela{#2}
  \def \posb{#3}
  \def \labelb{#4}
  \node at \posa (a) {\(\labela\)};
  \node at \posb (b) {\(\labelb\)};
  \draw [-angle 60] (a.east) .. controls +(30:1cm) and +(150:.5cm) .. 
                    (b.west);

}
