  % \disegno{\gnomino{0.7}{1.5}{f(a)}{a}{gray!50}{blue!50!black}{.69}}
  \def \x{#1}
  \def \y{#2}
  \def \lx{#3}
  \def \ly{#4}
  \def \colorline{#5}
  \def \colorpoint{#6}
  \def \lbelow{#7}

        (\x, 0) node [below=\lbelow] {\(\lx\)};
  \filldraw (\x, \y) [\colorpoint] circle(1.5pt);
}
