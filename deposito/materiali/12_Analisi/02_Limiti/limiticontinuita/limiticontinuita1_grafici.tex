\newcommand{\micx}[8]{% 
  % interno del microscopio posto sull'asse x.
  \def \xa{#1} \def \xb{#2} \def \xc{#3} \def \xd{#4}
  \def \ya{#5} \def \yb{#6} \def \yc{#7}
  \def \lab{#8}
  \draw (\xa, \ya) -- (\xd, \ya);
  \draw (\xa, \ya) -- (\xd, \ya);
  \draw [Green!50!black, ultra thick] (\xc, \yb) -- (\xb, \yb);
  \draw [Maroon!50!black, ultra thick] (\xb, \ya) node [below] {\lab} -- 
                                       (\xb, \yb);
  \draw (\xb, \yb) -- (\xb, \yc);
}
