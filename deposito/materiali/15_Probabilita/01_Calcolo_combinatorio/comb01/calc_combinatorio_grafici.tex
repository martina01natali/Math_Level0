\newcommand{\dadomonetacart}{%

  \def \miy{0}
  \def \may{7}

    \rcomvar{\mix}{\max}{\miy}{\may}{gray!50, very thin, step=1}{M}{D}
    \foreach \ux in {1, ..., 6} {
      \node at (\ux, 0) [below] {\(\ux\)};

        \fill (\ux, \uy) circle(2pt);}}
    \foreach \uy/\ly in {1/T, 2/C} {

  }
}
